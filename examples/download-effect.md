# Download Effect

Use a download effect when a provider action should start a browser download.
`download.route` is best for protected provider downloads because the widget
fetches the route with EmDash auth headers. `download.url` is best for public
files.

## Files Used

- `src/emdash/export-actions.ts`: define the provider plugin, manifest route,
  action route, and protected download route.
- `astro.config.mjs`: register `actionsPlugin()` and `exportActions()` in the
  EmDash plugin list. Add `export-actions` to `actionsPlugin({ providers })` if
  you want the dashboard to discover it too.
- Target collection schema file, for example `seed/schema.ts`,
  `seed/site.seed.json`, or the project-local script that defines EmDash
  collections. Add the field object to that collection's `fields` array.

## Backend Code

Create this provider module at `src/emdash/export-actions.ts`:

```ts
// src/emdash/export-actions.ts
import { definePlugin, type RouteContext } from "emdash";
import { defineActionsManifest, type ActionsManifest } from "@bnomei/emdash-actions";

export function exportActions() {
  return definePlugin({
    id: "export-actions",
    version: "0.1.0",
    capabilities: ["content:read"],
    routes: {
      actions: {
        handler: exportActionsManifestRoute,
      },
      "exports/latest": {
        handler: latestExportActionRoute,
      },
      "exports/latest.csv": {
        handler: latestExportDownloadRoute,
      },
    },
  });
}

async function exportActionsManifestRoute(): Promise<ActionsManifest> {
  return defineActionsManifest({
    actions: [
      {
        id: "exports.downloadLatest",
        label: "Download latest export",
        route: "exports/latest",
        method: "POST",
        placement: "field",
        icon: "copy",
        tone: "info",
      },
    ],
  });
}

async function latestExportActionRoute(ctx: RouteContext) {
  const serverTime = new Date().toISOString();
  await ctx.kv.set("state:lastRequestedAt", serverTime);

  return {
    ok: true,
    status: 200,
    severity: "success",
    message: `Export prepared on the server at ${serverTime}.`,
    effects: {
      download: {
        route: "exports/latest.csv",
        filename: "latest-export.csv",
      },
    },
  };
}

async function latestExportDownloadRoute(ctx: RouteContext) {
  const entries = ctx.content
    ? await ctx.content.list("posts", { limit: 50, orderBy: { updatedAt: "desc" } })
    : { items: [] };
  const generatedAt = new Date().toISOString();
  const csv = [
    "id,slug,status,updatedAt",
    ...entries.items.map((entry) =>
      [entry.id, entry.slug ?? "", entry.status, entry.updatedAt].map(csvCell).join(","),
    ),
  ].join("\n");

  ctx.log.info("Generated CSV export", {
    count: entries.items.length,
    generatedAt,
  });

  return new Response(`${csv}\n`, {
    headers: {
      "Content-Type": "text/csv; charset=utf-8",
      "Content-Disposition": "attachment; filename=latest-export.csv",
    },
  });
}

function csvCell(value: string) {
  return JSON.stringify(value);
}
```

## Field JSON

Put this object inside the target collection schema's `fields` array, for
example in a seed JSON file or a project-local schema setup script.

```json
{
  "slug": "export_action",
  "label": "Export Action",
  "type": "string",
  "widget": "actions:button",
  "options": {
    "mode": "run",
    "provider": "export-actions",
    "manifestRoute": "actions",
    "action": "exports.downloadLatest",
    "label": "Download latest export"
  }
}
```

## Frontend Behavior

The button calls `exports/latest`. The successful result tells the widget to
fetch `exports/latest.csv` from the same provider, create a temporary blob URL,
and click a hidden download link named `latest-export.csv`.

The CSV is generated by backend TypeScript. The provider declares
`capabilities: ["content:read"]`, queries EmDash content with `ctx.content`, and
returns a real `Response` object with CSV bytes.

For a public file, return this instead:

```json
{
  "effects": {
    "download": {
      "url": "/robots.txt",
      "filename": "robots.txt"
    }
  }
}
```
